140 // http://lolengine.net/blog/2013/07/27/rgb-to-hsv-in-glsl

142 vec3 rgb2hsv(vec3 c)
143 {
144  vec4 K = vec4(0.0, -1.0 / 3.0, 2.0 / 3.0, -1.0);
145  vec4 p = c.g < c.b ? vec4(c.bg, K.wz) : vec4(c.gb, K.xy);
146  vec4 q = c.r < p.x ? vec4(p.xyw, c.r) : vec4(c.r, p.yzx);
147  float d = q.x - min(q.w, q.y);
148  float e = 1.0e-10;
149  return vec3(abs(q.z + (q.w - q.y) / (6.0 * d + e)), d / (q.x + e), q.x);
150 }
151 
152 vec3 hsv2rgb(vec3 c)
153 {
154  vec4 K = vec4(1.0, 2.0 / 3.0, 1.0 / 3.0, 3.0);
155  vec3 p = abs(fract(c.xxx + K.xyz) * 6.0 - K.www);
156  return c.z * mix(K.xxx, clamp(p - K.xxx, 0.0, 1.0), c.y);
157 }

159 vec4 stereo_displace(vec4 v, float e, float r) {
160  vec3 OE = vec3(-v.z, 0.0, v.x); // eye direction, orthogonal to vertex vector
161  OE = normalize(OE); // but preserving +y up-vector
162  OE *= e; // set mag to eye separation
163  vec3 EV = v.xyz - OE; // eye to vertex
164  float ev = length(EV); // save length
165  EV /= ev; // normalize
166  // coefs for polynomial t^2 + 2bt + c = 0
167  // derived from cosine law r^2 = t^2 + e^2 + 2tecos(theta)
168  // where theta is angle between OE and EV
169  // t is distance to sphere surface from eye
170  float b = -dot(OE, EV); // multiply -1 to dot product because
171  // OE needs to be flipped in direction
172  float c = e * e - r * r;
173  float t = -b + sqrt(b * b - c); // quadratic formula
174  v.xyz = OE + t * EV; // direction from origin to sphere surface
175  v.xyz = ev * normalize(v.xyz); // normalize and set mag to eye-to-v distance
176  return v;
177 }
