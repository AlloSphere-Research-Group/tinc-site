165 vec3 rgb2hsv(vec3 c)
166 {
167  vec4 K = vec4(0.0, -1.0 / 3.0, 2.0 / 3.0, -1.0);
168  vec4 p = c.g < c.b ? vec4(c.bg, K.wz) : vec4(c.gb, K.xy);
169  vec4 q = c.r < p.x ? vec4(p.xyw, c.r) : vec4(c.r, p.yzx);
170  float d = q.x - min(q.w, q.y);
171  float e = 1.0e-10;
172  return vec3(abs(q.z + (q.w - q.y) / (6.0 * d + e)), d / (q.x + e), q.x);
173 }
174 
175 vec3 hsv2rgb(vec3 c)
176 {
177  vec4 K = vec4(1.0, 2.0 / 3.0, 1.0 / 3.0, 3.0);
178  vec3 p = abs(fract(c.xxx + K.xyz) * 6.0 - K.www);
179  return c.z * mix(K.xxx, clamp(p - K.xxx, 0.0, 1.0), c.y);
180 }

182 vec4 stereo_displace(vec4 v, float e, float r) {
183  vec3 OE = vec3(-v.z, 0.0, v.x); // eye direction, orthogonal to vertex vector
184  OE = normalize(OE); // but preserving +y up-vector
185  OE *= e; // set mag to eye separation
186  vec3 EV = v.xyz - OE; // eye to vertex
187  float ev = length(EV); // save length
188  EV /= ev; // normalize
189  // coefs for polynomial t^2 + 2bt + c = 0
190  // derived from cosine law r^2 = t^2 + e^2 + 2tecos(theta)
191  // where theta is angle between OE and EV
192  // t is distance to sphere surface from eye
193  float b = -dot(OE, EV); // multiply -1 to dot product because
194  // OE needs to be flipped in direction
195  float c = e * e - r * r;
196  float t = -b + sqrt(b * b - c); // quadratic formula
197  v.xyz = OE + t * EV; // direction from origin to sphere surface
198  v.xyz = ev * normalize(v.xyz); // normalize and set mag to eye-to-v distance
199  return v;
200 }

217  }

220  } else {
221  colormult = clipped_mult;
222  }
223  }
224 
225  float local_scale = 1.0;

228 
229  if (is_line > 0.5) {
230  local_scale = 1.03;
231  color = vec4(hsv2rgb(vec3(offset.w, 1.0, 1.0)), alpha)* colormult * reflectivity;
232  } else {

234  }

237 
238  if (is_omni > 0.5) {
239  gl_Position = al_ProjectionMatrix * stereo_displace(al_ModelViewMatrix * p, eye_sep, foc_len);

241  gl_Position = al_ProjectionMatrix * al_ModelViewMatrix * p;
242  }
