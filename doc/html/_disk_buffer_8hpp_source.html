96 template <class DataType>
97 bool DiskBuffer<DataType>::loadData(std::string filename, bool notify) {
98  std::unique_lock<std::mutex> lk(m_writeLock);
99  if (filename.size() > 0) {
100  m_fileName = filename;
101  }
102  std::shared_ptr<DataType> newData = BufferManager<DataType>::getWritable();
103  auto ret = parseFile(filename, newData);
104  if (ret) {
105  BufferManager<DataType>::doneWriting(newData);
106  }
107  lk.unlock();
108  for (auto cb : mUpdateCallbacks) {
109  cb(ret);
110  }
111  if (notify) {
112  for (auto cb : mNotifyCallbacks) {
113  cb(ret);
114  }
115  }
116  return ret;
117 }

122  auto fileName = getFilenameForWriting();
